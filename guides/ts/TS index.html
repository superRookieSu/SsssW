Notice:

  • any: let arr: any[] = [1, false, "0", {obj: true}]
  • none: means nothing. but typeof(null) ==> "object"
  • <type>value: let a = "1"; let b:number = <number> a;
  • value as type:
  • for (variable of object) ==> for each(variable in object) in ActionScript3.
  • function func_name(param1: string, param2?:string):void{}: [param?: type] == option param.
  • function func_name(param1: string, ...param2:string[]):void{}: [...param2: type[]] .
  • var a = [1, 2]; var [b, c] = a; // b = 1, c = 2;
  • var a: string|number; //means a can be string and number. Type1|Type2|...
  • var a: string[]|number[]; //means a can be string[] and number[]. Type1[]|Type2|[]...

Lambda function

  • var func = (x:number, y: string) => {dosth...} . normal lambda function
  • var func = x => {dosth...} . single param.
  • var func = () => {dosth...} . no param.

Duck Typing


        interface IPoint {
            x: number;
            y:number;
        }

        function addPoints(p1: IPoint, p1: IPoint): IPoint{
            return {x: p1.x + p2.x, y: p1.y + p2.y};
        }

        addPoints({x: 32, y: 43}, {x: 2, y: 3});    //right
        addPoints({x: 1}, {x: 4, y: 3});        //wrong;
